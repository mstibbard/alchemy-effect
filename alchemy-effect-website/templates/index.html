src/Api.ts Worker 
export default class Api extends Cloudflare.Worker<Api>()(
  "Api", { main: import.meta.path },
  Effect.gen(function* () {
    const rooms  = yield* Room;
    const agents = yield* Agent;
    return {
      fetch: Effect.gen(function* () {
        yield* rooms.getByName("lobby").broadcast("hello world");
        return HttpServerResponse.text(yield* agents.getByName("sandbox").hello());
      }),
    };
  }),
) {}

{# 2. Durable Object — hibernatable WebSockets #}
src/Room.ts Durable Object 
export default class Room extends Cloudflare.DurableObjectNamespace<Room>()(
  "Rooms", Effect.gen(function* () {
    return Effect.gen(function* () {
      const state = yield* Cloudflare.DurableObjectState;
      return {
        fetch: Effect.gen(function* () {
          const [res, ws] = yield* Cloudflare.upgrade();
          ws.serializeAttachment({ id: crypto.randomUUID() });  // survives hibernation
          return res;
        }),
        broadcast: (text: string) => Effect.gen(function* () {
          for (const ws of yield* state.getWebSockets()) yield* ws.send(text);
        }),
      };
    });
  }),
) {}

export class Sandbox extends Cloudflare.Container<Sandbox, {
  exec: (cmd: string) => Effect.Effect<{ exitCode: number; stdout: string }>;
}>()(
  "Sandbox", { main: import.meta.path, instanceType: "dev" },
) {}

// from a Durable Object — call into the running container
const container = yield* Cloudflare.runContainer(sandbox);
const { stdout } = yield* container.exec("cat /etc/os-release");
{# 4. Workflow #}
src/NotifyWorkflow.ts Workflow 
export default class NotifyWorkflow extends Cloudflare.Workflow<NotifyWorkflow>()(
  "Notifier", Effect.gen(function* () {
    const rooms = yield* Room;
    return Effect.gen(function* () {
      const { payload } = yield* Cloudflare.WorkflowEvent;
      yield* Cloudflare.task("process", Effect.succeed(payload.message));
      yield* Cloudflare.task("broadcast", rooms.getByName(payload.roomId).broadcast("hello"));
      yield* Cloudflare.sleep("cooldown", "2 seconds");
      yield* Cloudflare.task("finalize", rooms.getByName(payload.roomId).broadcast("done"));
    });
  }),
) {}

const loader = yield* Cloudflare.DynamicWorker("Loader");

const worker = loader.load({
  mainModule: "worker.js",
  modules: {
    "worker.js": `export default {
      async fetch(req) { return new Response(eval(await req.text())) }
    }`,
  },
});
return yield* worker.fetch(HttpClientRequest.post("https://worker/"));

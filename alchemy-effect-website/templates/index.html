{# ── How it works ── #}

How it works

  • Infrastructure-as-Effects — Your infrastructure and application logic live in the same program. Change one, the other stays in sync automatically.
  • Built on Effect — Type-safe errors, declarative retries, and streams — the same primitives you use for application code also power your infrastructure.
  • If it compiles, it deploys — Catch wiring mistakes at compile time, not after a five-minute deploy.
  • Local emulation — Run the full stack locally with alchemy dev. Real Workers, Durable Objects, and WebSockets — instant feedback without cloud roundtrips.
  • Live logs — Tail or search logs from any deployed resource. Give your agent the context it needs to diagnose and fix bugs.
  • OTEL "for free" — Logs, traces, and metrics are emitted automatically via Effect’s native OpenTelemetry support. No manual instrumentation.

src/Api.ts Worker 
export default class Api extends Worker<Api>()(
  "Api",
  {
    main: import.meta.path
  },
  Effect.gen(function* () {
    const chatRooms = yield* ChatRoom;
    return {
      fetch: Effect.gen(function* () {
        const room = chatRooms.getByName("lobby");
        yield* room.broadcast("hello");
        return HttpServerResponse.text("ok");
      }),
    };
  }),
) {}
src/ChatRoom.ts Durable Object 
export default class ChatRoom extends DurableObjectNamespace<ChatRoom>()(
  "ChatRooms",
  Effect.gen(function* () {
    return Effect.gen(function* () {
      const state = yield* DurableObjectState;
      return {
        fetch: Effect.gen(function* () {
          const [res, ws] = yield* upgrade();
          ws.serializeAttachment({
            id: crypto.randomUUID()
          });
          return res;
        }),
        broadcast: (text) =>
          state.getWebSockets().pipe(
            Effect.flatMap((sockets) =>
              Effect.all(sockets.map((ws) =>
                ws.send(text))),
            ),
          ),
      };
    });
  }),
) {}

  Sandbox,
  { exec: (cmd: string) => Effect.Effect<{
    exitCode: number; stdout: string
  }> }
>()(
  "Sandbox", { main: import.meta.path, instanceType: "dev" },
) {}

// from a Durable Object — call into the running container
const container = yield* Cloudflare.runContainer(sandbox);
const { stdout } = yield* container.exec("cat /etc/os-release");
{# Workflow #}
src/TransferWorkflow.ts Workflow 
export default class Transfer extends Workflow<Transfer>()(
  "Transfer",
  Effect.gen(function* () {
    const notifications = yield* Notifications;
    return Effect.gen(function* () {
      const { from, to, amount } =
        (yield* WorkflowEvent).payload;
      yield* task("debit", debitAccount(from, amount));
      yield* task("credit", creditAccount(to, amount));
      yield* sleep("settle", "30 seconds");
      yield* task("notify",
        notifications.send(from, to, amount));
    });
  }),
) {}
